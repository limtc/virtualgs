Moving Rectangle

 
 

program movingRect;

uses Types, QuickDraw, Events, MiscTool;


var

  x,y,speed: integer;

  tick: longInt;


begin

  x := 0;

  y := 100;

  tick := getTick;


  graphics(320);

  setRandSeed(tick);

  speed:=random mod 6+1;


  clearScreen(0);

  hideCursor;


  { Draw rectangle }


  setPenSize(5,3);

  setSolidPenPat(7);

  moveTo(x,y);

  line(0,0);


  repeat


    { Erase rectangle }


    setSolidPenPat(0);

    moveTo(x,y);

    line(0,0);


    x := x+speed;


    { Draw rectangle }


    setSolidPenPat(7);

    moveTo(x,y);

    line(0,0);


    if x > 320 then begin

      x := 0;

      sysBeep;

      speed := random mod 6+1;

      y := random mod 197;

    end;


    repeat until getTick > tick;

    tick := tick + 1;

  until button(0);

end.

This program draws a moving rectangle.


It demonstrate the animation technique.
